#### path 설정
    vi ~/.bashrc
    export PATH = $PATH:/home:/var
    export PATH="/root/.pyenv/bin:$PATH"
    eval "$(pyenv init -)"
    eval "$(pyenv virtualenv-init -)"
    source ~/.bashrc

#### pyenv 가상환경 만들기
    pyenv install --list
    pyenv install 3.6.4
    pyenv global 3.6.4
    pyenv versions

#### pyenv 가상환경 적용
    pyenv virtualenv 3.6.4 sample-env
    pyenv local sample-env
    pyenv shell uwsgi-env
